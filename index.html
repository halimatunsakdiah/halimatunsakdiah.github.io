function login() { var email = document.getElementById("loginEmail").value.trim().toLowerCase(); var password = document.getElementById("loginPassword").value.trim(); var savedEmail = localStorage.getItem("email"); var savedPassword = localStorage.getItem("password"); if (!savedEmail || !savedPassword) { alert("Silakan daftar terlebih dahulu!"); return; } if (email === savedEmail.toLowerCase() && password === savedPassword) { document.getElementById("navbar").style.display = "block"; showPage("home"); } else { alert("Email atau Password salah!"); } }
